    // botones utilitarios para la UI

    document.addEventListener('DOMContentLoaded', () => {

      const playBtn = document.getElementById('audio-play');

      const stopBtn = document.getElementById('audio-stop');

      const audioVideo = document.getElementById('audio-video');


      if (audioVideo) {

        // usar safeSetSrc si está disponible

        if (audioVideo.safeSetSrc) audioVideo.safeSetSrc('assets/media/test.mp4');

        else audioVideo.src = 'assets/media/test.mp4';

      }


      if (playBtn) playBtn.onclick = () => {

        if (window.safePlayVideo) window.safePlayVideo();

        else audioVideo.play().catch(()=>{});

      };

      if (stopBtn) stopBtn.onclick = () => {

        if (audioVideo) { audioVideo.pause(); audioVideo.currentTime = 0; }

      };


      // keyboard UI helpers

      const resetBtn = document.getElementById('keyboard-reset');

      const listBtn = document.getElementById('keyboard-list');

      if (resetBtn) resetBtn.onclick = () => { if (window.keyboardUtils) window.keyboardUtils.resetPressed(); };

      if (listBtn) listBtn.onclick = () => { if (window.keyboardUtils) console.log(window.keyboardUtils.getPressedKeys()); };


      // battery refresh

      const batteryRefresh = document.getElementById('battery-refresh');

      if (batteryRefresh) batteryRefresh.onclick = () => {

        if (window.initBattery) window.initBattery();

      }

    });
